\# --- CONFIG ---

CONF\_THRESH = 0.55 



\# SPEED OPTIMIZATION: 384x216 reduces math load by 20% compared to 416

AI\_W, AI\_H = 384, 216

class PerfectSyncApp:

&#x20;   def \_\_init\_\_(self):

&#x20;       print("--- Initializing Zero-Latency Engine ---")

&#x20;       core = ov.Core()

&#x20;       model = core.read\_model(MODEL\_XML)

&#x20;       

&#x20;       # We use LATENCY hint to make the CPU finish boxes as fast as possible

&#x20;       self.compiled\_model = core.compile\_model(model, "CPU", {

&#x20;           "PERFORMANCE\_HINT": "LATENCY",

&#x20;           "CACHE\_DIR": "./cache" # Speeds up subsequent startups

&#x20;       })

&#x20;       self.infer\_request = self.compiled\_model.create\_infer\_request()

&#x20;       

&#x20;       self.cap = cv2.VideoCapture(VIDEO\_SOURCE)

&#x20;       # Force the camera to not buffer old frames

&#x20;       self.cap.set(cv2.CAP\_PROP\_BUFFERSIZE, 1)



&#x20;       self.live\_frame = None

&#x20;       self.latest\_boxes = None

&#x20;       self.running = True



&#x20;   def capture\_thread(self):

&#x20;       """ 

&#x20;       High-speed capture thread. 

&#x20;       It 'grabs' frames and only 'retrieves' the very last one 

&#x20;       to ensure we never show old buffered video.

&#x20;       """

&#x20;       while self.running:

&#x20;           # Clear the internal buffer to get the 'Real-Present' frame

&#x20;           for \_ in range(2): 

&#x20;               self.cap.grab()

&#x20;           ret, frame = self.cap.retrieve()

&#x20;           

&#x20;           if ret:

&#x20;               self.live\_frame = frame

&#x20;           else:

&#x20;               time.sleep(0.01)



&#x20;   def ai\_thread(self):

&#x20;       """ 

&#x20;       Inference thread: Grabs the absolute latest frame 

&#x20;       the moment the CPU is ready for new math.

&#x20;       """

&#x20;       while self.running:

&#x20;           if self.live\_frame is not None:

&#x20;               # 1. Take the absolute current frame

&#x20;               raw\_img = self.live\_frame

&#x20;               h\_orig, w\_orig = raw\_img.shape\[:2]

&#x20;               

&#x20;               # 2. Pre-process (Fast resize)

&#x20;               img = cv2.resize(raw\_img, (AI\_W, AI\_H), interpolation=cv2.INTER\_LINEAR)

&#x20;               img = cv2.cvtColor(img, cv2.COLOR\_BGR2RGB)

&#x20;               img = img.transpose(2, 0, 1)

&#x20;               input\_tensor = np.expand\_dims(img, axis=0).astype(np.float32) / 255.0



&#x20;               # 3. Start Inference

&#x20;               self.infer\_request.infer({0: input\_tensor})

&#x20;               

&#x20;               # 4. Extract results

&#x20;               results = self.infer\_request.get\_output\_tensor(0).data # boxes

&#x20;               labels = self.infer\_request.get\_output\_tensor(1).data # labels

&#x20;               scores = self.infer\_request.get\_output\_tensor(2).data # scores

&#x20;               

&#x20;               # 5. Store with current scale

&#x20;               self.latest\_boxes = (results, labels, scores, w\_orig/AI\_W, h\_orig/AI\_H)

&#x20;           

&#x20;           # This sleep is tiny to ensure maximum AI cycles

&#x20;           time.sleep(0.001)



&#x20;   def run(self):

&#x20;       threading.Thread(target=self.capture\_thread, daemon=True).start()

&#x20;       threading.Thread(target=self.ai\_thread, daemon=True).start()

&#x20;       

&#x20;       print("🚀 SMOOTH \& SYNCED FEED ACTIVE. Press 'q' to quit.")

&#x20;       

&#x20;       while True:

&#x20;           if self.live\_frame is None: continue

&#x20;           

&#x20;           # THE LIVE VIDEO (Full Speed)

&#x20;           canvas = self.live\_frame.copy()



&#x20;           # THE BOXES (Latest available)

&#x20;           if self.latest\_boxes is not None:

&#x20;               boxes, labels, scores, x\_scale, y\_scale = self.latest\_boxes



&#x20;               for i in range(len(scores)):

&#x20;                   if scores\[i] > CONF\_THRESH:

&#x20;                       b = boxes\[i]

&#x20;                       x1, y1 = int(b\[0]\*x\_scale), int(b\[1]\*y\_scale)

&#x20;                       x2, y2 = int(b\[2]\*x\_scale), int(b\[3]\*y\_scale)

&#x20;                       

&#x20;                       label\_id = int(labels\[i])

&#x20;                       name = CLASS\_NAMES.get(label\_id, "Object")

&#x20;                       

&#x20;                       # Use a thin, bright green line for better look

&#x20;                       cv2.rectangle(canvas, (x1, y1), (x2, y2), (0, 255, 0), 2)

&#x20;                       cv2.putText(canvas, f"{name} {scores\[i]:.2f}", (x1, y1-8), 

&#x20;                                   cv2.FONT\_HERSHEY\_SIMPLEX, 0.45, (0, 255, 0), 2)



&#x20;           cv2.imshow("Capstone - Zero-Latency Optimized", canvas)

&#x20;           

&#x20;           # 1ms wait is enough for the UI and reduces display lag

&#x20;           if cv2.waitKey(1) \& 0xFF == ord('q'):

&#x20;               self.running = False

&#x20;               break

&#x20;       

&#x20;       self.cap.release()

&#x20;       cv2.destroyAllWindows()



if \_\_name\_\_ == "\_\_main\_\_":

&#x20;   PerfectSyncApp().run()

# ==============================================================================
# FINAL, SELF-CONTAINED SCRIPT FOR VIDEO INFERENCE DEMONSTRATION

import torch
import torchvision
from PIL import Image
import torchvision.transforms.v2 as T
import os
import cv2 # OpenCV library for video processing
print("--- All necessary libraries imported. ---")

# ==============================================================================
# STEP 1: DEFINE ALL NECESSARY HELPER FUNCTIONS
# These are required to build the model architecture and process the video frames.
# ==============================================================================
print("Defining helper functions...")

def get_transform(train):
    """
    Defines the data transforms. Must be IDENTICAL to the one used for training.
    """
    return T.Compose([
        T.Resize((360, 640)), 
        T.PILToTensor(), 
        T.ToDtype(torch.float32, scale=True)
    ])

def get_model(num_classes):
    """
    Builds the Faster R-CNN model architecture.
    """
    from torchvision.models.detection import FasterRCNN_ResNet50_FPN_Weights
    from torchvision.models.detection.faster_rcnn import FastRCNNPredictor
    
    # Load the pre-trained model structure
    model = torchvision.models.detection.fasterrcnn_resnet50_fpn(weights=FasterRCNN_ResNet50_FPN_Weights.DEFAULT)
    
    # Get the number of input features for the classifier
    in_features = model.roi_heads.box_predictor.cls_score.in_features
    
    # Replace the pre-trained head with a new one for our number of classes
    model.roi_heads.box_predictor = FastRCNNPredictor(in_features, num_classes)
    return model

# This dictionary maps the model's output index to a human-readable class name
CLASS_NAMES = {
    1: 'car', 2: 'bus', 3: 'person', 4: 'bike', 5: 'truck', 6: 'motor',
    7: 'train', 8: 'rider', 9: 'traffic sign', 10: 'traffic light'
}

print("Definitions complete.")

# ==============================================================================
# STEP 2: MAIN VIDEO INFERENCE SCRIPT
# ==============================================================================
def run_video_inference():
    # --- CONFIGURATION ---
    # A confidence of 0.5 is a good balance between seeing many boxes and only seeing accurate ones.
    CONFIDENCE_THRESHOLD = 0.5 

    # --- Paths ---
    # <<< ACTION REQUIRED >>> Update these with the folder and file names of your final model dataset
    MODEL_DATASET_FOLDER = 'bdd100k-fasterrcnn-v4'  # e.g., the name you gave your uploaded dataset
    MODEL_FILENAME = 'fasterrcnn_curriculum_final.pth' # The name of your best model file
    
    VIDEO_DATASET_FOLDER = 'driving-video-with-object-tracking'
    VIDEO_FILENAME = 'bdd100k_videos_train_00/bdd100k/videos/train/0000f77c-6257be58.mov'

    MODEL_PATH = f'/kaggle/input/{MODEL_DATASET_FOLDER}/{MODEL_FILENAME}'
    VIDEO_PATH = f'/kaggle/input/{VIDEO_DATASET_FOLDER}/{VIDEO_FILENAME}'
    OUTPUT_VIDEO_PATH = '/kaggle/working/output_video_final.mp4'
    # ---------------------

    # --- Verification ---
    if not os.path.exists(MODEL_PATH):
        print(f"❌ ERROR: Model file not found at '{MODEL_PATH}'")
        print("Please check that your 'MODEL_DATASET_FOLDER' and 'MODEL_FILENAME' variables are correct.")
        return
    if not os.path.exists(VIDEO_PATH):
        print(f"❌ ERROR: Video file not found at '{VIDEO_PATH}'")
        return

    print(f"✅ Model and Video found successfully.")
    device = torch.device('cuda') if torch.cuda.is_available() else torch.device('cpu')
    print(f"Using device: {device}")

    # --- Load Your Trained Model ---
    print("Loading your trained model...")
    model = get_model(num_classes=11)
    model.load_state_dict(torch.load(MODEL_PATH, map_location=device))
    model.to(device)
    model.eval()
    print("Model loaded successfully.")

    # --- Process the Video Frame by Frame ---
    cap = cv2.VideoCapture(VIDEO_PATH)
    frame_width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
    frame_height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
    fps = int(cap.get(cv2.CAP_PROP_FPS))
    
    fourcc = cv2.VideoWriter_fourcc(*'mp4v')
    out = cv2.VideoWriter(OUTPUT_VIDEO_PATH, fourcc, fps, (frame_width, frame_height))
    
    frame_count = 0
    print(f"\n--- Starting to process video: {os.path.basename(VIDEO_PATH)} ---")
    
    @torch.inference_mode()
    def process_frame(frame):
        # Convert frame from OpenCV (BGR) to PIL (RGB)
        image_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        image_pil = Image.fromarray(image_rgb)
        
        # Apply the same transforms as validation
        image_tensor = get_transform(train=False)(image_pil)
        image_tensor = image_tensor.to(device)
        
        # Get model prediction
        prediction = model([image_tensor])[0]
        
        # Draw boxes on the original frame
        for box, label, score in zip(prediction['boxes'], prediction['labels'], prediction['scores']):
            if score > CONFIDENCE_THRESHOLD:
                # Scale the box from the resized tensor size back to the original frame size
                box = box.cpu().numpy().astype(int)
                resized_h, resized_w = 360, 640
                x_scale = frame_width / resized_w
                y_scale = frame_height / resized_h
                
                x1, y1, x2, y2 = box
                x1, y1 = int(x1 * x_scale), int(y1 * y_scale)
                x2, y2 = int(x2 * x_scale), int(y2 * y_scale)
                
                class_name = CLASS_NAMES.get(label.cpu().item(), 'Unknown')
                
                # Draw rectangle and text using OpenCV
                cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)
                cv2.putText(frame, f"{class_name}: {score:.2f}", (x1, y1 - 10), 
                            cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
        return frame

    # Loop through the video
    while(cap.isOpened()):
        ret, frame = cap.read()
        if ret:
            processed_frame = process_frame(frame)
            out.write(processed_frame)
            frame_count += 1
            if frame_count % 30 == 0:
                print(f"  > Processed {frame_count} frames...")
        else:
            break
            
    # Release everything
    cap.release()
    out.release()
    
    print(f"\n--- FINISHED ---")
    print(f"Processed {frame_count} frames in total.")
    print(f"✅ Output video saved to: {OUTPUT_VIDEO_PATH}")
    print("You can now download this file from the '/kaggle/working/' directory in the right-hand menu.")

# Run the main function
if __name__ == '__main__':
    run_video_inference()
